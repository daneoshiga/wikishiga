Aliases do git
=

Aliases úteis para se manter no subgrupo [alias] do arquivo de configuração
~/.gitconfig

    [alias]
        a = add -p
        d = difftool
        st = status -sb
        ci = commit -v
        br = branch
        co = checkout
        df = diff -w --word-diff
        lg = log -p
        lol = log --graph --decorate --pretty=oneline --abbrev-commit
        lola = log --graph --decorate --pretty=oneline --abbrev-commit --all
